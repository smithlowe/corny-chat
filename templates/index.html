socket.on('message', d => { // 1. Identify who sent it const isMe = d.username === myN; // 2. Handle Profile Picture (Use robot if empty) const pic = d.profile_pic || `https://api.dicebear.com/7.x/bottts/svg?seed=${d.username}`; // 3. Handle Content (Text or Audio) let content = `${d.message}`; if(d.audio) { content = ``; } // 4. Create the HTML Bubble const html = `
${!isMe ? `${d.username}` : ''} ${content}
${d.time} ${isMe ? '✔' : ''}
`; // 5. Inject into Chat and Scroll const chatBox = document.getElementById('chat'); chatBox.insertAdjacentHTML('beforeend', html); chatBox.scrollTop = chatBox.scrollHeight; // 6. Mark as read if it's from someone else if (!isMe) { socket.emit('mark_read', { msg_id: d.id }); } });
